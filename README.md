# SQL SCHEMA:

Customer Schema:\

CREATE TABLE Customers(\
ID int NOT NULL AUTO_INCREMENT,\
NAME varchar(30) NOT NULL,\
ADDRESS Varchar(50) NOT NULL,\
MOBILE bigint NOT NULL,\
NOTE varchar(50),\
DUE int DEFAULT 0,\
PRIMARY KEY (ID)\
);

Transaction Schema:\

CREATE TABLE Transactions(\
ID int NOT NULL AUTO_INCREMENT,\
CID int NOT NULL,\
NAME varchar(30) NOT NULL,\
DATEE DATE NOT NULL,\
TYPE varchar(20) NOT NULL,\
QTY int,\
AMOUNT int NOT NULL,\
NOTE varchar(30),\
PRIMARY KEY (ID)\
);
